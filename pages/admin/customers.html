import { getCustomers } from "/services/admin/customers.js"; const searchInput = document.getElementById("search"); const searchBtn = document.querySelector(".search-button"); const tableBody = document.querySelector("#customers_table tbody"); let filters = {}; fetchCustomers(filters); // ===== Search ===== searchBtn.addEventListener("click", handleSearch); searchInput.addEventListener("keydown", (e) => { if (e.key === "Enter") handleSearch(); }); function handleSearch() { const query = searchInput.value.trim(); if (query.length > 0) { filters = { ...filters, query }; } else { delete filters.query; } fetchCustomers(filters); } // ===== Fetch customers ===== async function fetchCustomers(params = {}) { try { const response = await getCustomers(params); fillTotal(response.data.length); fillTable(response.data); } catch (error) { console.log("Customers error:", error); } } // ===== Fill total count ===== function fillTotal(count) { const span = document.querySelector("#total .value"); if (span) span.textContent = count; } // ===== Fill table ===== function fillTable(customers) { if (!customers.length) { tableBody.innerHTML = `No customers found`; return; } let body = document.createDocumentFragment(); customers.forEach((customer) => { const row = document.createElement("tr"); row.dataset.id = customer.id; row.innerHTML = ` ${customer.name} ${customer.email} ${customer.currently_borrowing} ${customer.borrowed} Show `; body.appendChild(row); }); tableBody.innerHTML = ""; tableBody.appendChild(body); }
